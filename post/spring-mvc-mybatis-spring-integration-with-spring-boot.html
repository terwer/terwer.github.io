                <include>**/**</include>
            </includes>
        </resource>
        <resource>
            <directory>src/main/resources</directory>

        </resource>

# йӣҶжҲҗLogbackпјҢз»ҹдёҖж—Ҙеҝ—жЎҶжһ¶

SpringBootй»ҳи®Өе·Із»ҸйӣҶжҲҗпјҢеҸӘйңҖиҰҒж·»еҠ й…ҚзҪ®ж–Үд»¶еҚіеҸҜпјҢеҰӮжһңдёҚж·»еҠ пјҢй»ҳи®Өд№ҹиғҪжү“еҚ°ж—Ҙеҝ—гҖӮ

 <configuration>
     <appender name="STDOUT" class="ch.qos.logback.core.ConsoleAppender">гҖҖгҖҖгҖҖгҖҖ
         <encoder>гҖҖгҖҖгҖҖгҖҖгҖҖгҖҖгҖҖгҖҖ
             <pattern>%-4relative [%thread] %-5level %logger{35} - %msg %n</pattern>гҖҖгҖҖгҖҖ
         </encoder>
     </appender>

     <!-- jetty -->
     <logger name="org.eclipse.jetty" level="ERROR"/>

     <root level="INFO">гҖҖгҖҖгҖҖ
         <appender-ref ref="STDOUT"/>гҖҖ
